public class YearRange {
    private int lowYear;
    private int highYear;

    public YearRange(int lowYear, int highYear) {
        this.lowYear = lowYear;
        this.highYear = highYear;
    }

    public int getLowYear() { return lowYear; }
    public int getHighYear() { return highYear; }

    public int getMidYear() {
        return Math.round((lowYear + highYear) / 2.0f);
    }

    public int getDifference() {
        return Math.abs(highYear - lowYear);
    }

    @Override
    public String toString() {
        return lowYear + "â€“" + highYear + " (mid " + getMidYear() + ")";
    }
}
