public class VinylRecord extends AuctionItem {
    private String albumName;
    private String artist;
    private String genre;
    private int diameter;

    public VinylRecord(String id, String albumName, String artist, String genre,
                       int diameter, YearRange yearRange, String owner,
                       String condition, double startingPrice)
    {
        super(id, yearRange, owner, condition, startingPrice);
        this.albumName = albumName;
        this.artist = artist;
        this.genre = genre;
        this.diameter = diameter;
    }
    public String getAlbumName() {
        return albumName;
    }

    public String getArtist() {
        return artist;
    }

    public String getGenre() {
        return genre;
    }

    public int getDiameter() {
        return diameter;
    }

    @Override
    public String shortDescription() {
        return String.format("%s - %s (%s, %d-inch)", artist, albumName, genre, diameter);
    }
}
