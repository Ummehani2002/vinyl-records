import javax.swing.SwingUtilities;

public class Main {
    public static void main(String[] args) {
        AuctionInventory inventory = new AuctionInventory();
        inventory.readFromCSV("collectibles.csv");

        // Launch the Swing GUI safely on the Event Dispatch Thread
        SwingUtilities.invokeLater(() -> new CollectibleGUI(inventory));
    }
}
