import java.io.*;
import java.util.*;
import java.util.stream.Collectors;

public class VinylCollection {
    private List<VinylRecord> records;

    public VinylCollection() {
        records = new ArrayList<>();
    }

    public void addRecord(VinylRecord record) {
        records.add(record);
    }

    public int getTotalRecords() {
        return records.size();
    }

    public VinylRecord getOldestRecord() {
        // Use the LOW year of the YearRange
        return records.stream()
                .min(Comparator.comparingInt(r -> r.getYearRange().getLowYear()))
                .orElse(null);
    }

    public VinylRecord getNewestRecord() {
        // Use the HIGH year of the YearRange
        return records.stream()
                .max(Comparator.comparingInt(r -> r.getYearRange().getHighYear()))
                .orElse(null);
    }

    public VinylRecord getMostExpensive() {
        return records.stream()
                .max(Comparator.comparingDouble(VinylRecord::getStartingPrice))
                .orElse(null);
    }

    public VinylRecord getLeastExpensive() {
        return records.stream()
                .min(Comparator.comparingDouble(VinylRecord::getStartingPrice))
                .orElse(null);
    }

    public Set<String> getUniqueOwners() {
        return records.stream()
                .map(VinylRecord::getOwner)
                .collect(Collectors.toSet());
    }

    // ✅ Read data from CSV file with error handling (Stage 3: includes low+high year)
    public void readFromCSV(String filePath) {
        System.out.println("Reading data from: " + filePath);
        int lineNum = 0;
        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = br.readLine()) != null) {
                lineNum++;
                String[] data = line.split(",");

                // Expected: id, album, artist, genre, diameter, lowYear, highYear, owner, condition, price
                if (data.length < 10) {
                    System.err.println("⚠️ Error (line " + lineNum + "): Missing fields → " + line);
                    continue;
                }

                try {
                    String id = data[0].trim();
                    String album = data[1].trim();
                    String artist = data[2].trim();
                    String genre = data[3].trim();
                    int diameter = Integer.parseInt(data[4].trim());
                    int lowYear = Integer.parseInt(data[5].trim());
                    int highYear = Integer.parseInt(data[6].trim());
                    String owner = data[7].trim();
                    String condition = data[8].trim();
                    double price = Double.parseDouble(data[9].trim());

                    // Validate condition
                    List<String> validConditions = Arrays.asList("Mint", "Restored", "Needs Restoring");
                    if (!validConditions.contains(condition)) {
                        System.err.println("⚠️ Error (line " + lineNum + "): Invalid condition → " + condition);
                        continue;
                    }

                    // Create YearRange
                    YearRange yr = new YearRange(lowYear, highYear);

                    // Create and store the record
                    VinylRecord record = new VinylRecord(id, album, artist, genre, diameter, yr, owner, condition, price);
                    records.add(record);

                } catch (NumberFormatException e) {
                    System.err.println("⚠️ Error (line " + lineNum + "): Invalid number → " + e.getMessage());
                }
            }
        } catch (IOException e) {
            System.err.println("❌ File read error: " + e.getMessage());
        }

        System.out.println("✅ Successfully loaded " + records.size() + " valid records.\n");
    }

    // ✅ Generate statistics summary and save to file
    public void writeSummary(String outputFile) {
        try (PrintWriter pw = new PrintWriter(new FileWriter(outputFile))) {
            pw.println("===== Vinyl Records Summary =====");
            pw.println("Total items: " + getTotalRecords());
            pw.println();

            VinylRecord oldest = getOldestRecord();
            VinylRecord newest = getNewestRecord();
            VinylRecord mostExp = getMostExpensive();
            VinylRecord leastExp = getLeastExpensive();

            pw.println("Oldest item: " + shortDesc(oldest));
            pw.println("Newest item: " + shortDesc(newest));
            pw.println("Most expensive: " + shortDesc(mostExp));
            pw.println("Least expensive: " + shortDesc(leastExp));
            pw.println();

            double avg = records.stream().mapToDouble(VinylRecord::getStartingPrice).average().orElse(0);
            double std = Math.sqrt(records.stream()
                    .mapToDouble(r -> Math.pow(r.getStartingPrice() - avg, 2))
                    .average().orElse(0));

            pw.printf("Average starting price: %.2f%n", avg);
            pw.printf("Standard deviation: %.2f%n%n", std);

            pw.println("Breakdown by condition:");
            Map<String, Long> condCount = records.stream()
                    .collect(Collectors.groupingBy(VinylRecord::getCondition, Collectors.counting()));
            for (Map.Entry<String, Long> e : condCount.entrySet()) {
                pw.printf("  %s: %d%n", e.getKey(), e.getValue());
            }

            pw.println("\nUnique owners:");
            for (String owner : getUniqueOwners()) {
                pw.println("  " + owner);
            }

        } catch (IOException e) {
            System.err.println("❌ Error writing summary: " + e.getMessage());
        }
    }

    private String shortDesc(VinylRecord r) {
        if (r == null) return "N/A";
        YearRange yr = r.getYearRange();
        return String.format("[%s] %s (%s, $%.2f)",
                r.getId(), r.getAlbumName(), yr.toString(), r.getStartingPrice());
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder("Vinyl Collection:\n");
        for (VinylRecord record : records) {
            sb.append(record.toString()).append("\n");
        }
        return sb.toString();
    }
}
