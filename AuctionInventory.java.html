import java.io.*;
import java.util.*;
import java.util.stream.Collectors;

public class AuctionInventory {
    private List<AuctionItem> items = new ArrayList<>();

    // ✅ Read from CSV
    public void readFromCSV(String filePath) {
        System.out.println("Reading: " + filePath);
        int lineNum = 0;
        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = br.readLine()) != null) {
                lineNum++;
                String[] d = line.split(",");

                if (d.length < 6) {
                    System.err.println("⚠️ Missing fields at line " + lineNum);
                    continue;
                }

                try {
                    String type = d[0].trim().toLowerCase();
                    String id = d[1].trim();
                    AuctionItem item = null;

                    switch (type) {
                        case "vinyl" -> {
                            String album = d[2], artist = d[3], genre = d[4];
                            int diameter = Integer.parseInt(d[5]);
                            int low = Integer.parseInt(d[6]), high = Integer.parseInt(d[7]);
                            YearRange yr = new YearRange(low, high);
                            String owner = d[8], cond = d[9];
                            double price = Double.parseDouble(d[10]);
                            item = new VinylRecord(id, album, artist, genre, diameter, yr, owner, cond, price);
                        }
                        case "furniture" -> {
                            String furnType = d[2], style = d[3], maker = d[4];
                            double len = Double.parseDouble(d[5]);
                            double height = Double.parseDouble(d[6]);
                            double depth = Double.parseDouble(d[7]);
                            int low = Integer.parseInt(d[8]), high = Integer.parseInt(d[9]);
                            YearRange yr = new YearRange(low, high);
                            String owner = d[10], cond = d[11];
                            double price = Double.parseDouble(d[12]);
                            item = new Furniture(id, furnType, style, maker, len, height, depth, yr, owner, cond, price);
                        }
                        case "sculpture" -> {
                            String subject = d[2], material = d[3];
                            double height = Double.parseDouble(d[4]);
                            int low = Integer.parseInt(d[5]), high = Integer.parseInt(d[6]);
                            YearRange yr = new YearRange(low, high);
                            String owner = d[7], cond = d[8];
                            double price = Double.parseDouble(d[9]);
                            item = new Sculpture(id, subject, material, height, yr, owner, cond, price);
                        }
                        case "coin" -> {
                            String material = d[2], origin = d[3], value = d[4];
                            int low = Integer.parseInt(d[5]), high = Integer.parseInt(d[6]);
                            YearRange yr = new YearRange(low, high);
                            String owner = d[7], cond = d[8];
                            double price = Double.parseDouble(d[9]);
                            item = new Coin(id, material, origin, value, yr, owner, cond, price);
                        }
                        default -> System.err.println("⚠️ Unknown type at line " + lineNum);
                    }

                    if (item != null) items.add(item);

                } catch (Exception e) {
                    System.err.println("⚠️ Error at line " + lineNum + ": " + e.getMessage());
                }
            }
        } catch (IOException e) {
            System.err.println("❌ File read error: " + e.getMessage());
        }

        System.out.println("✅ Loaded " + items.size() + " valid records.\n");
    }

    // ✅ Write summary report
    public void writeSummary(String filePath) {
        try (PrintWriter pw = new PrintWriter(new FileWriter(filePath))) {
            pw.println("===== Auction Inventory Summary =====");
            pw.println("Total items: " + items.size());

            AuctionItem oldest = items.stream()
                    .min(Comparator.comparingInt(i -> i.getYearRange().getLowYear()))
                    .orElse(null);
            AuctionItem newest = items.stream()
                    .max(Comparator.comparingInt(i -> i.getYearRange().getHighYear()))
                    .orElse(null);

            pw.println("\nOldest item: " + summaryLine(oldest));
            pw.println("Newest item: " + summaryLine(newest));

            List<AuctionItem> top3 = items.stream()
                    .sorted((a, b) -> Integer.compare(
                            b.getYearRange().getDifference(),
                            a.getYearRange().getDifference()))
                    .limit(3)
                    .toList();

            pw.println("\nTop 3 items with biggest year range:");
            for (AuctionItem i : top3)
                pw.println("  " + summaryLine(i));

            double avg = items.stream().mapToDouble(AuctionItem::getStartingPrice).average().orElse(0);
            double std = Math.sqrt(items.stream()
                    .mapToDouble(i -> Math.pow(i.getStartingPrice() - avg, 2))
                    .average().orElse(0));

            pw.printf("\nAverage starting price: %.2f%n", avg);
            pw.printf("Standard deviation: %.2f%n", std);

            Map<String, Long> condCount = items.stream()
                    .collect(Collectors.groupingBy(AuctionItem::getCondition, Collectors.counting()));
            pw.println("\nBreakdown by condition:");
            condCount.forEach((k, v) -> pw.printf("  %s: %d%n", k, v));

        } catch (IOException e) {
            System.err.println("❌ Error writing summary: " + e.getMessage());
        }
    }

    private String summaryLine(AuctionItem i) {
        if (i == null) return "N/A";
        return String.format("[%s] %s | %s | $%.2f",
                i.getId(), i.shortDescription(), i.getYearRange(), i.getStartingPrice());
    }

    // ✅ Getter for GUI use
    public List<AuctionItem> getItems() {
        return items;
    }

    // ✅ Save back to CSV (used by GUI Save button)
    public void saveToCSV(String filePath) {
        try (PrintWriter pw = new PrintWriter(new FileWriter(filePath))) {
            for (AuctionItem item : items) {
                YearRange yr = item.getYearRange();

                if (item instanceof VinylRecord vr) {
                    pw.printf("vinyl,%s,%s,%s,%s,%d,%d,%d,%s,%s,%.2f%n",
                            vr.getId(), vr.getAlbumName(), vr.getArtist(), vr.getGenre(),
                            vr.getDiameter(), yr.getLowYear(), yr.getHighYear(),
                            vr.getOwner(), vr.getCondition(), vr.getStartingPrice());
                } else if (item instanceof Furniture f) {
                    pw.printf("furniture,%s,%s,%s,%s,%.1f,%.1f,%.1f,%d,%d,%s,%s,%.2f%n",
                            f.getId(), f.getType(), f.getStyle(), f.getMaker(),
                            f.getLength(), f.getHeight(), f.getDepth(),
                            yr.getLowYear(), yr.getHighYear(),
                            f.getOwner(), f.getCondition(), f.getStartingPrice());
                } else if (item instanceof Sculpture s) {
                    pw.printf("sculpture,%s,%s,%s,%.1f,%d,%d,%s,%s,%.2f%n",
                            s.getId(), s.getSubject(), s.getMaterial(), s.getHeight(),
                            yr.getLowYear(), yr.getHighYear(),
                            s.getOwner(), s.getCondition(), s.getStartingPrice());
                } else if (item instanceof Coin c) {
                    pw.printf("coin,%s,%s,%s,%s,%d,%d,%s,%s,%.2f%n",
                            c.getId(), c.getMaterial(), c.getOrigin(), c.getOriginalValue(),
                            yr.getLowYear(), yr.getHighYear(),
                            c.getOwner(), c.getCondition(), c.getStartingPrice());
                }
            }
            System.out.println("✅ Saved all items to " + filePath);
        } catch (IOException e) {
            System.err.println("❌ Error writing CSV: " + e.getMessage());
        }
    }
}
