public abstract class AuctionItem {
    private String id;
    private YearRange yearRange;
    private String owner;
    private String condition;
    private double startingPrice;

    public AuctionItem(String id, YearRange yearRange, String owner, String condition, double startingPrice) {
        this.id = id;
        this.yearRange = yearRange;
        this.owner = owner;
        this.condition = condition;
        this.startingPrice = startingPrice;
    }

    public String getId() { return id; }
    public YearRange getYearRange() { return yearRange; }
    public String getOwner() { return owner; }
    public String getCondition() { return condition; }
    public double getStartingPrice() { return startingPrice; }

    public abstract String shortDescription();

    @Override
    public String toString() {
        return String.format("[%s] %s | Year: %s | Owner: %s | Condition: %s | Start: $%.2f",
                id, getClass().getSimpleName(), yearRange, owner, condition, startingPrice);
    }
    public void setCondition(String condition) {
        this.condition = condition;
    }

    public void setStartingPrice(double startingPrice) {
        this.startingPrice = startingPrice;
    }
}
