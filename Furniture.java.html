public class Furniture extends AuctionItem {
    private String type;
    private String style;
    private String maker;
    private double length;
    private double height;
    private double depth;

    public Furniture(String id, String type, String style, String maker,
                     double length, double height, double depth,
                     YearRange yearRange, String owner, String condition, double startingPrice) {
        super(id, yearRange, owner, condition, startingPrice);
        this.type = type;
        this.style = style;
        this.maker = maker;
        this.length = length;
        this.height = height;
        this.depth = depth;
    }

    @Override
    public String shortDescription() {
        return String.format("%s %s (%s)", style, type, maker);
    }

    // âœ… Add these getters
    public String getType() { return type; }
    public String getStyle() { return style; }
    public String getMaker() { return maker; }
    public double getLength() { return length; }
    public double getHeight() { return height; }
    public double getDepth() { return depth; }
}
