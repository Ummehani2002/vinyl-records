public class Coin extends AuctionItem {
    private String material;
    private String origin;
    private String originalValue;

    public Coin(String id, String material, String origin, String originalValue,
                YearRange yearRange, String owner, String condition, double startingPrice) {
        super(id, yearRange, owner, condition, startingPrice);
        this.material = material;
        this.origin = origin;
        this.originalValue = originalValue;
    }

    @Override
    public String shortDescription() {
        return String.format("%s coin from %s (value %s)", material, origin, originalValue);
    }

    // âœ… Add getters for saveToCSV()
    public String getMaterial() { return material; }
    public String getOrigin() { return origin; }
    public String getOriginalValue() { return originalValue; }
}
