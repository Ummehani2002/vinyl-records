CollectibleGUI.java 
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.util.List;
import java.util.ArrayList;
import java.util.Comparator;

public class CollectibleGUI extends JFrame {
    private AuctionInventory inventory;
    private DefaultListModel<AuctionItem> listModel;
    private JList<AuctionItem> itemList;

    public CollectibleGUI(AuctionInventory inventory) {
        this.inventory = inventory;

        setTitle("Collectible Inventory System");
        setSize(800, 450);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new BorderLayout(10, 10));

        // ----- Item List -----
        listModel = new DefaultListModel<>();
        for (AuctionItem item : inventory.getItems()) {
            listModel.addElement(item);
        }

        itemList = new JList<>(listModel);
        itemList.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
        add(new JScrollPane(itemList), BorderLayout.CENTER);

        // ----- Button Panel -----
        JPanel btnPanel = new JPanel();
        btnPanel.setLayout(new FlowLayout(FlowLayout.CENTER, 10, 10));

        JButton moreInfo = new JButton("More Info");
        JButton edit = new JButton("Edit");
        JButton sortID = new JButton("Sort by ID");
        JButton sortPrice = new JButton("Sort by Price");
        JButton sortYear = new JButton("Sort by Year");
        JButton summary = new JButton("Generate Summary");
        JButton save = new JButton("Save CSV");
        JButton exit = new JButton("Exit");

        btnPanel.add(moreInfo);
        btnPanel.add(edit);
        btnPanel.add(sortID);
        btnPanel.add(sortPrice);
        btnPanel.add(sortYear);
        btnPanel.add(summary);
        btnPanel.add(save);
        btnPanel.add(exit);

        add(btnPanel, BorderLayout.SOUTH);

        // ----- Button Actions -----

        // Show full details
        moreInfo.addActionListener(e -> showItemDetails());

        // Edit condition and price
        edit.addActionListener(e -> editItem());

        // Sorting buttons
        sortID.addActionListener(e -> sortItems(Comparator.comparing(AuctionItem::getId)));
        sortPrice.addActionListener(e -> sortItems(Comparator.comparingDouble(AuctionItem::getStartingPrice)));
        sortYear.addActionListener(e -> sortItems(Comparator.comparingInt(i -> i.getYearRange().getMidYear())));

        // Generate summary file
        summary.addActionListener(e -> {
            inventory.writeSummary("summary.txt");
            JOptionPane.showMessageDialog(this, "✅ Summary file written successfully!", "Summary", JOptionPane.INFORMATION_MESSAGE);
        });

        // Save updates back to CSV
        save.addActionListener(e -> {
            inventory.saveToCSV("collectibles.csv");
            JOptionPane.showMessageDialog(this, "✅ Data saved to collectibles.csv", "Save Successful", JOptionPane.INFORMATION_MESSAGE);
        });

        // Exit
        exit.addActionListener(e -> System.exit(0));

        setVisible(true);
    }

    // ----- Sort Helper -----
    private void sortItems(Comparator<AuctionItem> comparator) {
        List<AuctionItem> sorted = new ArrayList<>(inventory.getItems());
        sorted.sort(comparator);
        listModel.clear();
        for (AuctionItem item : sorted) {
            listModel.addElement(item);
        }
    }

    // ----- More Info -----
    private void showItemDetails() {
        AuctionItem selected = itemList.getSelectedValue();
        if (selected == null) {
            JOptionPane.showMessageDialog(this, "Please select an item first.");
            return;
        }
        JOptionPane.showMessageDialog(
                this,
                selected.toString(),
                "Item Details",
                JOptionPane.INFORMATION_MESSAGE
        );
    }

    // ----- Edit Item -----
    private void editItem() {
        AuctionItem selected = itemList.getSelectedValue();
        if (selected == null) {
            JOptionPane.showMessageDialog(this, "Select an item to edit.");
            return;
        }

        try {
            String newCondition = JOptionPane.showInputDialog(this, "Enter new condition:", selected.getCondition());
            if (newCondition == null || newCondition.isBlank()) return;

            String newPriceStr = JOptionPane.showInputDialog(this, "Enter new starting price:", selected.getStartingPrice());
            if (newPriceStr == null || newPriceStr.isBlank()) return;

            double newPrice = Double.parseDouble(newPriceStr);
            selected.setCondition(newCondition);
            selected.setStartingPrice(newPrice);

            // Refresh the list display
            itemList.repaint();
            JOptionPane.showMessageDialog(this, "✅ Item updated successfully!");

        } catch (NumberFormatException ex) {
            JOptionPane.showMessageDialog(this, "⚠️ Invalid price format. Please enter a number.");
        }
    }
}
